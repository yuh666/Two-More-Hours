class Solution {

    private int[] ww;
    private int max;

    public Solution(int[] w) {
        int[] ww = new int[w.length];
        ww[0] = w[0];
        for (int i = 1; i < w.length; i++) {
            ww[i] = ww[i - 1] + w[i];
        }
        this.ww = ww;
        this.max = ww[ww.length - 1];
    }

    public int pickIndex() {
        int randomNum = new Random().nextInt(max) + 1;
        return binarySearch(randomNum);
    }

    private int binarySearch(int randomNum) {
        int lo = 0;
        int hi = ww.length - 1;
        while (lo < hi) {
            int mid = (lo + hi) / 2;
            if (ww[mid] == randomNum) {
                return mid;
            }
            if (ww[mid] < randomNum) {
                lo = mid + 1;
            } else {
                hi = mid;
            }
        }
        return hi;
    }
}

 * Solution obj = new Solution(w);
 * int param_1 = obj.pickIndex();
