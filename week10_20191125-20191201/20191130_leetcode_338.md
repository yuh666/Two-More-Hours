class Solution {
    public int[] countBits(int num) {
        int[] result = new int[num + 1];
        for(int i = 1; i <= num; i++){
            // 奇数Bits = 上一个偶数Bits + 1
            if((i & 1) == 1){
                result[i] = result[i-1] + 1;
            }
            // 偶数Bits = 除以2之后的偶数Bits
            else{
                result[i] = result[i>>1];
            }
        }
        return result;
    }
}
