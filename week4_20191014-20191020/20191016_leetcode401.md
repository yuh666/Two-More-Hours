class Solution {
    private static int[] nums = {1, 2, 4, 8, 1, 2, 4, 8, 16, 32};
    private int[] numSelected = new int[nums.length];
    private int num = 0;

    public List<String> readBinaryWatch(int num) {
        if (num <= 0)
            return Collections.singletonList("0:00");
        this.num = num;
        List<String> allWatch = new ArrayList<>();
        for (int i = 0; i < nums.length; i++) {
            numSelected[i] = 1;
            readBinaryWatch(i, 1, allWatch);
            numSelected[i] = 0;
        }
        return allWatch;
    }

    private void readBinaryWatch(int selectIndex, int step, List<String> allWatch) {
        if (step == num) {
            String watch = watch();
            if (watch != null)
                allWatch.add(watch);
            return;
        }
        for (int i = selectIndex + 1; i < nums.length; i++) {
            numSelected[i] = 1;
            readBinaryWatch(i, step + 1, allWatch);
            numSelected[i] = 0;
        }
    }

    private String watch() {
        int hour = 0;
        int minute = 0;
        for (int i = 0; i < 4; i++) {
            if (numSelected[i] == 1)
                hour += nums[i];
        }
        if (hour >= 12)
            return null;
        for (int i = 4; i < numSelected.length; i++) {
            if (numSelected[i] == 1)
                minute += nums[i];
        }
        if (minute >= 60)
            return null;

        return hour + ":" + (minute < 10 ? ("0" + minute) : minute);
    }
}
