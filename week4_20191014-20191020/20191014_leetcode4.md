class Solution {
    public double findMedianSortedArrays(int[] nums1, int[] nums2) {
        int[] mergeNums = merge(nums1, nums2);

        int leftIndex = (nums1.length + nums2.length - 1) / 2;
        int rightIndex = (nums1.length + nums2.length) / 2;
        return (mergeNums[leftIndex] + mergeNums[rightIndex]) / 2D;
    }

    private int[] merge(int[] nums1, int[] nums2) {
        int[] mergeNums = new int[nums1.length + nums2.length];
        int nums1Index = 0;
        int nums2Index = 0;
        int mergeNumsIndex = 0;
        while (nums1Index < nums1.length && nums2Index < nums2.length) {
            if (nums1[nums1Index] <= nums2[nums2Index]) {
                mergeNums[mergeNumsIndex++] = nums1[nums1Index];
                nums1Index++;
            } else {
                mergeNums[mergeNumsIndex++] = nums2[nums2Index];
                nums2Index++;
            }
        }

        while (nums1Index < nums1.length) {
            mergeNums[mergeNumsIndex++] = nums1[nums1Index];
            nums1Index++;
        }

        while (nums2Index < nums2.length) {
            mergeNums[mergeNumsIndex++] = nums2[nums2Index];
            nums2Index++;
        }

        return mergeNums;
    }
}
