class Solution {
    public String multiply(String num1, String num2) {
        if (num1 == null || num2 == null || num1.length() == 0 || num2.length() == 0)
            return null;

        int[] numResult = new int[num1.length() + num2.length()];
        for (int i = num1.length() - 1; i >= 0; i--) {
            for (int j = num2.length() - 1; j >= 0; j--) {
                numResult[i + j + 1] += (num1.charAt(i) - '0') * (num2.charAt(j) - '0');
            }
        }

        int jw = 0;
        for (int i = numResult.length - 1; i >= 0; i--) {
            int nowNum = numResult[i];
            int afterNum = (nowNum + jw) % 10;
            jw = (nowNum + jw) / 10;
            numResult[i] = afterNum;
        }

        int firstNonZero = 0;
        for (; firstNonZero < numResult.length-1; firstNonZero++)
            if (numResult[firstNonZero] != 0)
                break;

        String result = "";
        for (int i = firstNonZero; i < numResult.length; i++)
            result += numResult[i];

        return result;
    }
}
