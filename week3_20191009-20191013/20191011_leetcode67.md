class Solution {
    public String addBinary(String a, String b) {
        if (a == null || a.isEmpty() || b == null || b.isEmpty())
            return null;
        char[] aChars = a.toCharArray();
        char[] bChars = b.toCharArray();
        int aLength = aChars.length;
        int bLength = bChars.length;
        int maxLength = Math.max(aLength, bLength);

        boolean isJw = false;
        StringBuilder result1 = new StringBuilder();
        for (int i = 1; i <= maxLength; i++) {
            char aChar = aLength - i < 0 ? '0' : aChars[aLength - i];
            char bChar = bLength - i < 0 ? '0' : bChars[bLength - i];
            if (aChar == '1' && bChar == '1') {
                result1.insert(0, isJw ? "1" : "0");
                isJw = true;
            } else if ((aChar == '1' && bChar == '0') || (aChar == '0' && bChar == '1')) {
                result1.insert(0, isJw ? "0" : "1");
            } else {
                result1.insert(0, isJw ? "1" : "0");
                isJw = false;
            }
        }
        if (isJw)
            result1.insert(0, "1");
        return result1.toString();
    }
}
