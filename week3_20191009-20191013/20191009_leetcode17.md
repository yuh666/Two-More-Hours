class Solution {
    private char[] two = {'a', 'b', 'c'};
    private char[] three = {'d', 'e', 'f'};
    private char[] four = {'g', 'h', 'i'};
    private char[] five = {'j', 'k', 'l'};
    private char[] six = {'m', 'n', 'o'};
    private char[] seven = {'p', 'q', 'r', 's'};
    private char[] eight = {'t', 'u', 'v'};
    private char[] nine = {'w', 'x', 'y', 'z'};

    public List<String> letterCombinations(String digits) {
        List<String> list = new ArrayList();
        char[] chars = digits.toCharArray();
        backtrack("", chars, 0, list);

        return list;
    }

    private void backtrack(String combination, char[] digits, int digitsIndex, List<String> list) {
        if (digitsIndex == digits.length) {
            if(combination.length() != 0)
                list.add(combination);
            return;
        }
        char[] chars = chars(digits[digitsIndex]);
        for (int i = 0; i < chars.length; i++) {
            backtrack(combination + chars[i], digits, digitsIndex + 1, list);
        }
    }

    private char[] chars(char num) {
        switch (num) {
            case '2':
                return two;
            case '3':
                return three;
            case '4':
                return four;
            case '5':
                return five;
            case '6':
                return six;
            case '7':
                return seven;
            case '8':
                return eight;
            case '9':
                return nine;
        }
        return new char[]{};
    }
}
