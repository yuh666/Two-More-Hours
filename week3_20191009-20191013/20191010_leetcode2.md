class Solution {
    public ListNode addTwoNumbers(ListNode l1, ListNode l2) {
        int num1 = l1.val;
        int num2 = l2.val;
        int sum = num1 + num2;
        int now = sum % 10;
        // 需要进位的数
        int jw = (sum - now) / 10;

        ListNode root = new ListNode(now);
        ListNode prev = root;

        while (l1.next != null || l2.next != null || jw != 0) {
            l1 = l1.next == null ? new ListNode(0) : l1.next;
            l2 = l2.next == null ? new ListNode(0) : l2.next;

            num1 = l1.val;
            num2 = l2.val;
            sum = num1 + num2 + jw;
            now = sum % 10;
            ListNode otherNode = new ListNode(now);
            prev.next = otherNode;
            prev = otherNode;

            jw = (sum - now) / 10;

        }

        return root;
    }

    // 本地测试用
    public ListNode listNode(long num) {
        int firstVal = (int) (num - num / 10 * 10);
        ListNode root = new ListNode(firstVal);
        ListNode prev = root;

        for (num /= 10; num > 0; num /= 10) {
            int otherVal = (int) (num - num / 10 * 10);
            ListNode otherNode = new ListNode(otherVal);
            prev.next = otherNode;
            prev = otherNode;
        }
        return root;
    }
}
