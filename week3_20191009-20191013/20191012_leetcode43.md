class Solution {
    public String multiply(String num1, String num2) {
        if (num1 == null || num2 == null || num1.length() == 0 || num2.length() == 0)
            return null;
        char[] num2Chars = num2.toCharArray();
        String result = "";
        for (int i = num2Chars.length - 1; i >= 0; i--) {
            String resultl0 = multiply(num1, num2Chars[i]);
            int addZero = num2Chars.length - 1 - i;
            while (addZero > 0) {
                resultl0 += "0";
                addZero--;
            }
            result = add(result, resultl0);
        }
        int checkIndex = 0;
        while (result.length() > 1) {
            if (result.charAt(0) == '0')
                result = result.substring(1);
            else
                break;
        }
        return result;
    }

    private String add(String num1, String num2) {
        int maxLength = Math.max(num1.length(), num2.length());
        int backIndex = 0;
        int jw = 0;
        String result = "";
        while (backIndex < maxLength) {
            int num1_ = backIndex < num1.length() ? (num1.charAt(num1.length() - 1 - backIndex) - 48) : 0;
            int num2_ = backIndex < num2.length() ? (num2.charAt(num2.length() - 1 - backIndex) - 48) : 0;
            int result0 = num1_ + num2_ + jw;
            int result1 = result0 % 10;
            jw = result0 / 10;
            result = result1 + result;
            backIndex++;
        }
        if (jw != 0)
            result = jw + result;
        return result;
    }

    private String multiply(String num1, char num2) {
        int jw = 0;
        char[] num1Chars = num1.toCharArray();
        String result = "";
        for (int i = num1Chars.length - 1; i >= 0; i--) {
            int result0 = (num1Chars[i] - 48) * (num2 - 48) + jw;
            int result1 = result0 % 10;
            jw = result0 / 10;
            result = result1 + result;
        }
        if (jw != 0)
            result = jw + result;
        return result;
    }
}
