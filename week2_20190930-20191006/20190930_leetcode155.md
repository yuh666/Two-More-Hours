class MinStack {
    
    private int top = -1;
    private int[] nums;
    private int[] mins;

    public MinStack() {
        nums = new int[16];
        mins = new int[16];
    }

    public void push(int x) {
        if (top == nums.length - 1) {
            nums = Arrays.copyOf(nums, nums.length << 1);
            mins = Arrays.copyOf(mins, mins.length << 1);
        }
        top++;
        nums[top] = x;
        mins[top] = (top == 0 || x < mins[top - 1] ) ? x : mins[top - 1];
    }

    public void pop() {
        top--;
    }

    public int top() {
        return nums[top];
    }

    public int getMin() {
        return mins[top];
    }
}
