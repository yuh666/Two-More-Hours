class MinStack {
    
    private int nowMin = Integer.MAX_VALUE;
    private int stackHeight = 0;
    private Stack<Integer> stack;
    private Map<Integer, Integer> historyMinMap;
    

    /** initialize your data structure here. */
    public MinStack() {
        stack = new Stack();
        historyMinMap = new HashMap();
        historyMinMap.put(stackHeight, nowMin);
    }
    
    public void push(int x) {
        stack.push(x);
        if(x < nowMin)
            nowMin = x;
        historyMinMap.put(++stackHeight, nowMin);
    }
    
    public void pop() {
        int x = stack.pop();
        nowMin = historyMinMap.get(--stackHeight);
    }
    
    public int top() {
        return stack.peek();
    }
    
    public int getMin() {
        return nowMin;
    }
}
