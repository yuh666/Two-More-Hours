class MinStack {
    
    private int top = -1;
    private int length;
    private int[] nums;
    private int[] mins;

    public Solution155_2() {
        this.length = 16;
        this.nums = new int[this.length];
        this.mins = new int[this.length];
    }

    public void push(int x) {
        if (top == length - 1) {
            int newLength = length << 1;
            nums = Arrays.copyOf(nums, newLength);
            mins = Arrays.copyOf(mins, newLength);
            length = newLength;
        }
        top++;
        nums[top] = x;
        mins[top] = (top == 0 || x < mins[top - 1]) ? x : mins[top - 1];
    }

    public void pop() {
        top--;
    }

    public int top() {
        return nums[top];
    }

    public int getMin() {
        return mins[top];
    }
}
