package leetcode;

import java.util.ArrayList;
import java.util.HashMap;
import java.util.List;
import java.util.Map;
import java.util.function.Function;

public class Solution102 {

    public static void main(String[] args) {
        // [3,9,20,null,null,15,7]
        TreeNode three = new TreeNode(3);
        three.left = new TreeNode(9);
        three.right = new TreeNode(20);
        three.right.left = new TreeNode(15);
        three.right.right = new TreeNode(7);

        Solution102 solution102 = new Solution102();
        solution102.levelOrder(three);
    }


    public List<List<Integer>> levelOrder(TreeNode root) {
        Map<Integer, List<Integer>> map = new HashMap<>();
        traverse(root, 0, map);
        return new ArrayList<>(map.values());
    }

    private void traverse(TreeNode node, int level, Map<Integer, List<Integer>> map) {
        if (node == null)
            return;

        List<Integer> integers = map.computeIfAbsent(level, new Function<Integer, List<Integer>>() {
            @Override
            public List<Integer> apply(Integer integer) {
                return new ArrayList<>();
            }
        });

        integers.add(node.val);
        traverse(node.left, level + 1, map);
        traverse(node.right, level + 1, map);
    }

    public static class TreeNode {
        int val;
        TreeNode left;
        TreeNode right;

        TreeNode(int x) {
            val = x;
        }
    }

}
