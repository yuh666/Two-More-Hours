class Solution {
    public int strStr(String haystack, String needle) {
        if (needle == null || needle.length() == 0 ) {
            return 0;
        }
        if (needle.length() > haystack.length()) {
            return -1;
        }
        char[] haystackChars = haystack.toCharArray();
        char[] needleChars = needle.toCharArray();
        char first = needleChars[0];
        boolean onlyOne = needle.length() == 1;
        // haystackChars.length - i >= needleChars.length 这一步判断，非常重要，直接将耗时干到1ms
        for (int i = 0; i < haystackChars.length && haystackChars.length - i >= needleChars.length; i++) {
            if (haystackChars[i] != first)
                continue;
            if (onlyOne)
                return i;
            int j = 1;
            for (; j < needleChars.length && i + j < haystackChars.length; j++)
                if (haystackChars[i + j] != needleChars[j])
                    break;
            if (j == needleChars.length) {
                return i;
            }
        }
        return -1;
    }
}
