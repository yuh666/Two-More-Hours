class Solution {
    public boolean isMatch(String s, String p) {
        return match(s, p, 0, 0);
    }

    public boolean match(String s, String p, int si, int pi) {
        if (pi == p.length()) {
            return si == s.length();
        }
        if (pi + 1 < p.length() && p.charAt(pi + 1) == '*') {
            if (si < s.length() && (s.charAt(si) == p.charAt(pi) || p.charAt(pi) == '.')) {
                return match(s, p, si + 1, pi) || match(s, p, si, pi + 2);
            } else {
                return match(s, p, si, pi + 2);
            }
        } else if (si < s.length() && (s.charAt(si) == p.charAt(pi) || p.charAt(pi) == '.')) {
            return match(s, p, si + 1, pi + 1);
        }
        return false;
    }
}
