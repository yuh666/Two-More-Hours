class Solution {
    public List<List<Integer>> threeSum(int[] nums) {
        List<List<Integer>> resultList = new ArrayList<>();
        Arrays.sort(nums);
        for (int left = 0; left < nums.length - 2; left++) {
            if (left > 0 && nums[left] == nums[left - 1])
                continue;
            int now = left + 1;
            int right = nums.length - 1;
            while (now < right) {
                if (nums[left] + nums[now] + nums[right] < 0) {
                    for (now = now + 1; now < right; now++)
                        if (nums[now] != nums[now - 1])
                            break;
                } else if (nums[left] + nums[now] + nums[right] > 0) {
                    for (right = right - 1; right > now; right--)
                        if (nums[right] != nums[right + 1])
                            break;
                } else {
                    resultList.add(Arrays.asList(nums[left], nums[now], nums[right]));
                    for (right = right - 1; right > now; right--)
                        if (nums[right] != nums[right + 1])
                            break;
                }
            }
        }
        return resultList;
    }
}
