class RandomizedSet {
    private List<Integer> data;
    private Map<Integer, Integer> dataIndex;

    public RandomizedSet() {
        data = new ArrayList<>();
        dataIndex = new HashMap<>();
    }

    public boolean insert(int val) {
        if (dataIndex.containsKey(val))
            return false;

        data.add(val);
        dataIndex.put(val, data.size() - 1);

        return true;
    }

    public boolean remove(int val) {
        if (!dataIndex.containsKey(val))
            return false;

        Integer lastNum = data.get(data.size() - 1);
        
        Integer valIndex = dataIndex.remove(val);
        data.set(valIndex, lastNum);
        dataIndex.put(lastNum, valIndex);

        data.remove(data.size() - 1);
        return true;
    }

    public int getRandom() {
        if (data.isEmpty())
            return 0;
        if (data.size() == 1)
            return data.get(0);
        int randomIndex = (int) (Math.random() * 10 * (data.size() - 1));
        return data.get(randomIndex % (data.size() - 1));
    }
}
