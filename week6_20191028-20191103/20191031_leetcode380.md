class RandomizedSet {
    private ArrayList<Integer> data;
    private HashMap<Integer, Integer> dataIndex;

    public RandomizedSet() {
        data = new ArrayList<>();
        dataIndex = new HashMap<>();
    }

    public boolean insert(int val) {
        if (dataIndex.containsKey(val))
            return false;

        data.add(val);
        dataIndex.put(val, data.size() - 1);

        return true;
    }

    public boolean remove(int val) {
        if (!dataIndex.containsKey(val))
            return false;

        Integer valIndex = dataIndex.remove(val);
        Integer lastNum = data.remove(data.size() - 1);

        if (lastNum != val) {
            data.set(valIndex, lastNum);
            dataIndex.put(lastNum, valIndex);
        }

        return true;
    }

    public int getRandom() {
        if (data.isEmpty())
            return 0;
        if (data.size() == 1)
            return data.get(0);
        int randomIndex = (int) (Math.random() * data.size());
        return data.get(randomIndex);
    }
}
