class WordDictionary {

    private Node root = new Node();

    public void addWord(String word) {
        if (word == null || word.isEmpty())
            throw new RuntimeException("word is empty!");

        Node prevNode = root;

        char[] chars = word.toCharArray();
        for (char aChar : chars) {
            Node now = prevNode.childNodes[aChar - 'a'];
            if (now == null) {
                now = new Node(aChar);
                prevNode.childNodes[aChar - 'a'] = now;
            }
            prevNode = now;
        }

        prevNode.isWord = true;
    }

    public boolean search(String word) {
        if (word == null || word.isEmpty())
            throw new RuntimeException("word is empty!");
        return search_(word, root);
    }

    private boolean search_(String word, Node prevNode) {
        if (word.isEmpty())
            return prevNode.isWord;

        char firstChar = word.charAt(0);
        if (firstChar == '.') {
            for (Node childNode : prevNode.childNodes) {
                if (childNode != null && search_(word.substring(1), childNode)) {
                    return true;
                }
            }
            return false;
        }

        Node node = prevNode.childNodes[firstChar - 'a'];
        if (node == null)
            return false;
        return search_(word.substring(1), node);
    }

    public WordDictionary() {

    }

    private class Node {

        char value;
        boolean isWord;
        Node[] childNodes = new Node[26];

        private Node(char value) {
            this.value = value;
        }

        private Node() {
        }
    }
}
