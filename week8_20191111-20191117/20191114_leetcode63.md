class Solution {
    public int uniquePathsWithObstacles(int[][] obstacleGrid) {
        if(obstacleGrid.length == 0 || obstacleGrid[0].length == 0){
            return 0;
        }
        if(obstacleGrid[0][0] == 1 || obstacleGrid[obstacleGrid.length - 1][obstacleGrid[0].length - 1] == 1){
            return 0;
        }
        obstacleGrid[0][0] = 1;
        for(int i2 = 1; i2 < obstacleGrid[0].length; i2++){
            if(obstacleGrid[0][i2] == 1){
                obstacleGrid[0][i2] = 0;
                continue;
            }
            obstacleGrid[0][i2] = obstacleGrid[0][i2 - 1] == 0 ? 0 : 1;
        }
        for(int i1 = 1; i1 < obstacleGrid.length; i1++){
            if(obstacleGrid[i1][0] == 1){
                obstacleGrid[i1][0] = 0;
                continue;
            }
            obstacleGrid[i1][0] = obstacleGrid[i1 - 1 ][0] == 0 ? 0 : 1;
        }
        for(int i1 = 1; i1 < obstacleGrid.length; i1++){
            for(int i2 = 1; i2 < obstacleGrid[0].length; i2++){
                if(obstacleGrid[i1][i2] == 1){
                    obstacleGrid[i1][i2] = 0;
                    continue;
                }
                obstacleGrid[i1][i2] = obstacleGrid[i1 - 1][i2] + obstacleGrid[i1][i2 - 1];
            }
        }
        return obstacleGrid[obstacleGrid.length - 1][obstacleGrid[obstacleGrid.length - 1].length - 1];
    }
}
