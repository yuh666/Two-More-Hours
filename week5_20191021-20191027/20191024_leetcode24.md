/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 * 替换 now 和 next
 * left | now next | right
 */
class Solution {
    public ListNode swapPairs(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode newHead = head.next;
        ListNode left = null;
        ListNode now = head;
        while (now != null && now.next != null) {
            ListNode next = now.next;
            ListNode right = now.next.next;
            // 交换
            if (left != null)
                left.next = next;
            now.next = right;
            next.next = now;
            // 重置now
            left = now;
            now = right;
        }
        return newHead;
    }
}
