package leetcode;

public class Leetcode50 {
    // 1、特殊情况
    // 2、二分思想
    public double myPow(double x, int n) {
        if (n == 0)
            return 1;
        if (n == 1)
            return x;
        if (n == -1)
            return 1 / x;
        double sub0Result = myPow(x, n / 2);
        double sub1Result = myPow(x, n % 2);
        double result = 0.0;
        if (n % 2 == 0)
            result = sub0Result * sub0Result;
        else
            result = sub0Result * sub0Result * sub1Result;
        return result;
    }

    public static void main(String[] args) {
        Leetcode50 leetcode50 = new Leetcode50();
        double v = leetcode50.myPow(34.00515, -3);
        System.out.println(v);
    }
}
