/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 * https://leetcode.wang/leetcode-142-Linked-List-CycleII.html
 */
public class Solution {
    public ListNode detectCycle(ListNode head) {
        ListNode slow = head;
        ListNode fast = head;

        while (fast != null) {
            if (fast.next == null)
                return null;
            slow = slow.next;
            fast = fast.next.next;

            // 相遇节点
            if (slow == fast) {
                ListNode left = head;
                ListNode right = slow;
                while (left != right) {
                    left = left.next;
                    right = right.next;
                }
                return left;
            }
        }
        return null;
    }
}
