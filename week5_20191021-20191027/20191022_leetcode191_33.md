public class Solution {
    // you need to treat n as an unsigned value
    public int hammingWeight(int n) {
        int num = 0;
        while(n != 0){
            num+=(n & 1);
            n >>>= 1;     
        }
        return num;
    }
}

package leetcode;

public class Leetcode33 {

    public int search(int[] nums, int target) {
        if (nums == null || nums.length == 0)
            return -1;

        int rotationIndex = rotationIndex(nums, 0, nums.length - 1);
        if (target == nums[0]) {
            return 0;
        } else if (target == nums[rotationIndex]) {
            return rotationIndex;
        } else if (target == nums[nums.length - 1]) {
            return nums.length - 1;
        }

        if (target < nums[nums.length - 1]) {
            return binarySearch(nums, rotationIndex + 1, nums.length - 1, target);
        } else if (target < nums[rotationIndex]) {
            return binarySearch(nums, 0, rotationIndex, target);
        } else {
            return -1;
        }
    }

    private int binarySearch(int[] nums, int left, int right, int target) {
        if (left > right)
            return -1;
        if (left == right)
            return nums[left] == target ? left : -1;
        int mid = (left + right) / 2;
        if (nums[mid] > target) {
            return binarySearch(nums, left, mid, target);
        } else if (nums[mid] < target) {
            return binarySearch(nums, mid + 1, right, target);
        } else {
            return mid;
        }
    }

    private int rotationIndex(int[] nums, int left, int right) {
        if (left >= right)
            return left;
        int mid = (left + right) / 2;
        if (nums[mid] > nums[right]) {
            if (mid + 1 == right)
                return mid;
            return rotationIndex(nums, mid, right);
        } else {
            return rotationIndex(nums, left, mid);
        }
    }

    public static void main(String[] args) {
        int[] nums = new int[]{3, 5, 1};
        int target = 5;

        Leetcode33 leetcode33 = new Leetcode33();
        int search = leetcode33.search(nums, target);
        System.out.println(search);
    }
}
