class Solution {
    public int lengthOfLongestSubstring(String s) {
        if (s == null || s.length() == 0)
            return 0;
        if (s.length() == 1)
            return 1;

        int strStart = 0;
        int strEnd = 0;
        int longest = 0;
        Map<Character, Integer> charIndex = new HashMap();
        charIndex.put(s.charAt(0), 0);
        for (int i = 1; i < s.length(); i++) {
            if (charIndex.containsKey(s.charAt(i)) && charIndex.get(s.charAt(i)) >= strStart) {
                strStart = charIndex.get(s.charAt(i)) + 1;
            }
            strEnd++;
            charIndex.put(s.charAt(i), i);
            int now = strEnd - strStart + 1;
            longest = Math.max(longest, now);
        }
        return longest;
    }
}
