class MyQueue {

    private Stack<Integer> last = new Stack<>();
    private Stack<Integer> first = new Stack<>();

    public MyQueue() {

    }


    public void push(int x) {
        while (!first.isEmpty())
            last.push(first.pop());

        last.push(x);
    }


    public int pop() {
        while (!last.isEmpty())
            first.push(last.pop());

        return first.pop();
    }


    public int peek() {
        while (!last.isEmpty())
            first.push(last.pop());

        return first.peek();
    }


    public boolean empty() {
        return last.isEmpty() && first.isEmpty();
    }
}
