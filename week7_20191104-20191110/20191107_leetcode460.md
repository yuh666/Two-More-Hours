package leetcode.L460;

import java.util.Arrays;

public class LFUCache {

    private int[] container;
    private int size;
    private int capacity;

    public LFUCache(int capacity) {
        this.container = new int[capacity + 1];
        this.capacity = capacity + 1;
        this.size = 0;
    }

    public int get(int key) {
        return -1;
    }

    public void put(int key, int value) {
        if (size < capacity) {
            container[++size] = key;
            shiftUp(size);
        }
    }

    private void shiftUp(int index) {
        while (index > 1 && index <= size) {
            int parent = container[index / 2];
            if (parent < container[index]) {
                break;
            }
            int changeIndex = index / 2;
            change(changeIndex, index);
            index = changeIndex;
        }
    }

    private void shiftDown(int index) {
        while (index <= size / 2) {
            int left = container[index * 2];
            int right = (index * 2 + 1 <= size) ? container[index * 2 + 1] : Integer.MAX_VALUE;
            int changeIndex = (left < right) ? (index * 2) : (index * 2 + 1);
            if (container[changeIndex] > container[index]) {
                break;
            }
            change(changeIndex, index);
            index = changeIndex;
        }
    }

    private void change(int aIndex, int bIndex) {
        int temp = container[aIndex];
        container[aIndex] = container[bIndex];
        container[bIndex] = temp;
    }


    public LFUCache(int[] arr) {
        this.container = arr;
        this.size = container.length - 1;
    }

    public void heapify() {
        for (int i = size / 2; i > 0; i--) {
            shiftDown(i);
        }
    }

    public int[] getContainer() {
        return container;
    }

    public static void main(String[] args) {
        int[] oldArr = {0, 1, 5, 9, 3, 4, 5, 8, 6, 6, 4, 6};
        int[] newArr = {1, 5, 9, 3, 4, 5, 8, 6, 6, 4, 6};
        LFUCache lfuCache = new LFUCache(newArr.length);

        for (int i : newArr) {
            lfuCache.put(i, i);
        }

        int[] container = lfuCache.getContainer();
        System.out.println(Arrays.toString(container));

    }
}
