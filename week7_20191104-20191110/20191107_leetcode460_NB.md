package com.lizx.leetcode.L460;


import java.util.HashMap;
import java.util.Map;
import java.util.TreeSet;

public class LFUCache {

    /**
     * 存储所有的数据
     * key -> key
     * value -> Data
     */
    private Map<Integer, Data> dataMap;
    /**
     * 存储所有的桶
     * 一个桶只能放使用率相同的数据
     * key -> useCount
     * value -> Bucket
     */
    private Map<Integer, Bucket> bucketMap;
    /**
     * 所有不为空的bucket
     */
    private TreeSet<Integer> useCountSet;
    /**
     * 缓存容量
     */
    private int capacity;
    /**
     * 元素数量
     */
    private int size;

    public LFUCache(int capacity) {
        this.dataMap = new HashMap<>();
        this.bucketMap = new HashMap<>();
        this.useCountSet = new TreeSet<>();
        this.capacity = capacity;
        this.size = 0;
    }

    public int get(int key) {
        Data data = dataMap.get(key);
        if (data == null) {
            return -1;
        }

        // 从原来的桶中删除
        Bucket bucket = bucketMap.get(data.useCount);
        bucket.remove(key);
        if (bucket.isEmpty()) {
            useCountSet.remove(data.useCount);
        }

        data.useCount++;

        // 加到新的桶中
        Bucket newBucket = bucketMap.computeIfAbsent(data.useCount, Bucket::new);
        newBucket.add(key);
        useCountSet.add(data.useCount);

        return data.value;
    }

    public void put(int key, int value) {
        if (capacity <= 0) {
            return;
        }

        Data data = dataMap.get(key);
        if (data != null) {
            // 从原bucket中删除
            Bucket bucket = bucketMap.get(data.useCount);
            bucket.remove(key);
            if (bucket.isEmpty()) {
                useCountSet.remove(data.useCount);
            }

            data.useCount++;
            data.value = value;

            // 加入到新的bucket
            Bucket newBucket = bucketMap.computeIfAbsent(data.useCount, Bucket::new);
            newBucket.add(key);
            useCountSet.add(data.useCount);
            return;
        }

        // 删除使用率最小并且最近没有访问的
        if (size >= capacity) {
            Integer minUseCount = useCountSet.first();
            Bucket bucket = bucketMap.get(minUseCount);
            int removeKey = bucket.remove();
            dataMap.remove(removeKey);
            if (bucket.isEmpty()) {
                useCountSet.remove(minUseCount);
            }
            size--;
        }

        data = new Data(key, value, 0);
        dataMap.put(key, data);
        // 添加元素到对应的bucket
        Bucket bucket = bucketMap.computeIfAbsent(0, Bucket::new);
        bucket.add(key);
        useCountSet.add(0);
        size++;
    }

    /**
     * 相同使用率的元素都在一个bucket中
     * 数据结构：双向链表
     * 元素顺序：按照访问顺序排序；访问一个元素时放(移)在最后，head之外的第一个节点是最近没有访问的
     */
    private static class Bucket {
        Map<Integer, Node> keyMap;
        Node head;
        Node tail;

        Bucket(int useCount) {
            this.head = new Node(-1);
            this.tail = new Node(-1);
            this.keyMap = new HashMap<>();
            head.next = tail;
            tail.prev = head;
        }

        void add(int key) {
            Node node = keyMap.get(key);
            if (node != null) {
                // 从原来节点删除
                Node prev = node.prev;
                Node next = node.next;
                prev.next = next;
                next.prev = prev;
            } else {
                node = new Node(key);
            }

            // 添加到链表尾部
            Node tailPrev = tail.prev;
            tailPrev.next = node;
            node.next = tail;
            tail.prev = node;
            node.prev = tailPrev;

            keyMap.put(key, node);
        }

        void remove(int key) {
            Node node = keyMap.get(key);
            if (node == null) {
                throw new RuntimeException(key + " 不存在");
            }
            Node prev = node.prev;
            Node next = node.next;
            prev.next = next;
            next.prev = prev;

            keyMap.remove(key);
        }

        /**
         * 移除最近未访问的元素
         */
        int remove() {
            if (head.next == tail) {
                throw new RuntimeException("bucket is empty");
            }

            Node removeNode = head.next;
            Node removeNodeNext = removeNode.next;

            head.next = removeNodeNext;
            removeNodeNext.prev = head;

            keyMap.remove(removeNode.key);
            return removeNode.key;
        }

        boolean isEmpty() {
            return keyMap.isEmpty();
        }
    }

    private static class Data {
        int key;
        int value;
        int useCount;

        Data(int key, int value, int useCount) {
            this.key = key;
            this.value = value;
            this.useCount = useCount;
        }
    }

    private static class Node {
        int key;
        Node prev;
        Node next;

        Node(int key) {
            this.key = key;
        }
    }
