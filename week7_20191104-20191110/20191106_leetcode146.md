class LRUCache {

    private Node head = new Node();
    private Node tail = new Node();
    private Map<Integer, Node> map;
    private int capacity;
    private int size = 0;


    public LRUCache(int capacity) {
        this.capacity = capacity;
        this.map = new HashMap<>(capacity * 2);
        this.head.next = this.tail;
    }

    public int get(int key) {
        Node now = map.get(key);
        if (now == null) {
            return -1;
        }

        moveToFirst(now);

        return now.val;
    }

    private void moveToFirst(Node now) {
        removeNode(now);
        addToFirst(now);
    }

    private void removeNode(Node now) {
        Node prev = now.prev;
        Node next = now.next;
        prev.next = next;
        next.prev = prev;
    }

    private void addToFirst(Node now) {
        Node headNext = head.next;
        head.next = now;
        now.next = headNext;
        headNext.prev = now;
        now.prev = head;
    }


    public void put(int key, int value) {
        Node now = map.get(key);
        if (now != null) {
            now.val = value;
            moveToFirst(now);
            return;
        }
        now = new Node(key, value);
        if (size < capacity) {
            size++;
            addToFirst(now);
        } else {
            map.remove(tail.prev.key);
            removeNode(tail.prev);
            addToFirst(now);
        }
        map.put(key, now);
    }


    class Node {
        int key;
        int val;
        Node prev;
        Node next;

        private Node() {
        }

        private Node(int key, int val) {
            this.key = key;
            this.val = val;
        }
    }
}

 * Your LRUCache object will be instantiated and called as such:

 * int param_1 = obj.get(key);
