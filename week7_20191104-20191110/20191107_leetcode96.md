class Solution {
    
    public int numTrees(int n) {
        if(n == 0 || n == 1 || n == 2){
            return n;
        }
        int[] result = new int[n + 1];
        result[0] = 1;
        result[1] = 1;
        result[2] = 2;
        for(int i = 3; i <= n; i++){
            int totalNum = i - 1;
            for(int leftNum = 0; leftNum <= totalNum/2; leftNum++){
                int rightNum = totalNum - leftNum;
                int tempSum = result[leftNum] * result[rightNum];
                result[i] += tempSum;
                if(leftNum != rightNum){
                    result[i] += tempSum;
                }
            }
        }
        return result[n];
    }
}
