class MyStack {

    private LinkedList<Integer> queue = new LinkedList<>();

    public MyStack() {

    }

    public void push(int x) {
        queue.addLast(x);
    }

    public int pop() {
        if (queue.isEmpty())
            return -1;
        return queue.pollLast();
    }

    public int top() {
        if (queue.isEmpty())
            return -1;
        return queue.peekLast();
    }

    public boolean empty() {
        return queue.isEmpty();
    }
}
